  <script>

    async function updateStatus() {

      const sheetId = "1dM1CJWLBETKHOWi3Ieq-aNiCSGc9OqnLS0a0nIgoqzk";

      const url = `https://docs.google.com/spreadsheets/d/${sheetId}/gviz/tq?tqx=out:json`;

      const res = await fetch(url);

      const text = await res.text();

      const json = JSON.parse(text.substr(47).slice(0, -2));


      const rows = json.table.rows;

      const lastRow = rows[rows.length - 1];

      const status = lastRow.c[1].v;

      const timestamp = new Date(lastRow.c[0].v);


      const isWorking = status.toLowerCase().includes("funkar");


      document.body.style.backgroundColor = isWorking ? "green" : "red";

      document.getElementById("status").textContent =

        `Senaste rapport: ${status} – ${timestamp.toLocaleString("sv-SE")}`;

    }


    window.onload = updateStatus;

  </script>
